    95→function getToolGradient(name: string) {
    96→  const gradients = [
    97→    "from-violet-500 to-purple-600",
    98→    "from-blue-500 to-indigo-600",
    99→    "from-emerald-500 to-teal-600",
   100→    "from-orange-500 to-red-500",
   101→    "from-pink-500 to-rose-600",
   102→    "from-cyan-500 to-blue-600",
   103→    "from-amber-500 to-yellow-500",
   104→    "from-fuchsia-500 to-purple-600",
   105→    "from-lime-500 to-green-600",
   106→    "from-sky-500 to-indigo-500",
   107→    "from-red-500 to-orange-500",
   108→    "from-teal-500 to-cyan-600",
   109→  ];
   110→  let hash = 0;
   111→  for (let i = 0; i < name.length; i++) hash = name.charCodeAt(i) + ((hash << 5) - hash);
   112→  return gradients[Math.abs(hash) % gradients.length];
   113→}

  1051→function ToolIcon({ name, url, small = false }: { name: string; url?: string; small?: boolean }) {
  1052→  const gradient = getToolGradient(name);
  1053→  const domain = url ? extractDomain(url) : null;
  1054→  const cacheKey = domain || "";
  1055→  const cached = cacheKey ? logoCache.get(cacheKey) : undefined;
  1056→
  1057→  // "logo" = clearbit worked, "favicon" = clearbit failed but google favicon works, "failed" = both failed
  1058→  const [source, setSource] = useState<"loading" | "logo" | "favicon" | "failed">(
  1059→    cached === "logo" ? "logo" : cached === "favicon" ? "favicon" : cached === "failed" ? "failed" : domain ? "loading" : "failed"
  1060→  );
  1061→
  1062→  const logoUrl = domain ? `https://logo.clearbit.com/${domain}` : null;
  1063→  const faviconUrl = domain ? `https://www.google.com/s2/favicons?domain=${domain}&sz=64` : null;
  1064→
  1065→  const handleLogoLoaded = useCallback(() => {
  1066→    if (cacheKey) logoCache.set(cacheKey, "logo");
  1067→    setSource("logo");
  1068→  }, [cacheKey]);
  1069→
  1070→  const handleLogoError = useCallback(() => {
  1071→    // Clearbit logo failed — try Google favicon next
  1072→    if (cacheKey) logoCache.set(cacheKey, "favicon");
  1073→    setSource("favicon");
  1074→  }, [cacheKey]);
  1075→
  1076→  const handleFaviconLoaded = useCallback(() => {
  1077→    setSource("favicon");
  1078→  }, []);
  1079→
  1080→  const handleFaviconError = useCallback(() => {
  1081→    if (cacheKey) logoCache.set(cacheKey, "failed");
  1082→    setSource("failed");
  1083→  }, [cacheKey]);
  1084→
  1085→  const sizeClass = small ? "size-8" : "size-10";
  1086→
  1087→  // Real brand logo from Clearbit
  1088→  if (logoUrl && (source === "loading" || source === "logo")) {
  1089→    return (
  1090→      <span
  1091→        className={`shrink-0 overflow-hidden rounded-lg border border-border/50 ${sizeClass} ${
  1092→          source === "loading" ? `grid place-items-center bg-gradient-to-br ${gradient} border-transparent` : "bg-white dark:bg-zinc-800"
  1093→        }`}
  1094→      >
  1095→        <img
  1096→          src={logoUrl}
  1097→          alt={name}
  1098→          width={small ? 32 : 40}
  1099→          height={small ? 32 : 40}
  1100→          loading="lazy"
  1101→          onLoad={handleLogoLoaded}
  1102→          onError={handleLogoError}
  1103→          className="size-full object-contain p-0.5"
  1104→          style={{ display: source === "logo" ? "block" : "none" }}
  1105→        />
  1106→        {source === "loading" && (
  1107→          <span className={`font-bold text-white ${small ? "text-[9px]" : "text-xs"}`}>
  1108→            {initials(name)}
  1109→          </span>
  1110→        )}
  1111→      </span>
  1112→    );
  1113→  }
  1114→
  1115→  // Fallback: Google favicon (larger, but not a full logo)
  1116→  if (faviconUrl && source === "favicon") {
  1117→    return (
  1118→      <span
  1119→        className={`shrink-0 overflow-hidden rounded-lg border border-border/50 bg-white dark:bg-zinc-800 ${sizeClass}`}
  1120→      >
  1121→        <img
  1122→          src={faviconUrl}
  1123→          alt={name}
  1124→          width={small ? 32 : 40}
  1125→          height={small ? 32 : 40}
  1126→          loading="lazy"
  1127→          onLoad={handleFaviconLoaded}
  1128→          onError={handleFaviconError}
  1129→          className="size-full object-contain p-0.5"
  1130→        />
  1131→      </span>
  1132→    );
  1133→  }
  1134→
  1135→  // Final fallback: gradient initials
  1136→  return (
  1137→    <span
  1138→      className={`grid shrink-0 place-items-center rounded-lg bg-gradient-to-br ${gradient} font-bold text-white shadow-sm ${
  1139→        small ? "size-8 text-[9px]" : "size-10 text-xs"
  1140→      }`}
  1141→    >
  1142→      {initials(name)}
  1143→    </span>
  1144→  );
  1145→}

  1227→function BackToTop() {
  1228→  const [show, setShow] = useState(false);
  1229→  useEffect(() => {
  1230→    const handler = () => setShow(window.scrollY > 400);
  1231→    window.addEventListener("scroll", handler, { passive: true });
  1232→    return () => window.removeEventListener("scroll", handler);
  1233→  }, []);
  1234→  if (!show) return null;
  1235→  return (
  1236→    <button
  1237→      onClick={() => window.scrollTo({ top: 0, behavior: "smooth" })}
  1238→      className="fixed bottom-6 right-6 z-40 grid size-10 place-items-center rounded-full border border-border bg-card shadow-lg transition hover:bg-accent"
  1239→      aria-label="Back to top"
  1240→    >
  1241→      <ChevronRight className="size-4 -rotate-90" />
